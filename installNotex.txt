## make mqttio always run
# install systemd to manage running processes
sudo apt-get install systemd

# copy service file
sudo cp mqttio.service /etc/systemd/system/mqttio.service

# enable service
sudo systemctl enable mqttio.service
sudo systemctl start mqttio.service

# to check status of service 
watch systemctl status mqttio.service
